const SamplePage: React.FC = () => {
  //@DESC: separate file for styling
  const style = useStyles();

  //@DESC: separate file for functions
  const { handleDecrement, handleIncrement } = useCounterHook();

  return (
    <Box sx={style.root}>

    </Box>
  );
};
